library(Lahman)
library(dplyr)
library(ggplot2)
df<-Master%>%
  filter(!is.na(weight))%>%
  select(weight)
ggplot()+
  geom_histogram(data=df,aes(x=weight),color="white",fill="blue",bins=40)+
  ggtitle("Baseball Player Weight Distribution")
